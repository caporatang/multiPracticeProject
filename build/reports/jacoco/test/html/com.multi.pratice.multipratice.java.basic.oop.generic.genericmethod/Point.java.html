public class Point<T, V> {
    // 제네릭메서드를 활용해서 좌표를 받아 사각형의 면적을 구해보자!
    T x;
    V y;

    Point(T x, V y) {
        this.x = x;
        this.y = y;
    }
    public T getX() {
        return x;
    }
    public V getY() {
        return y;
    }
}
